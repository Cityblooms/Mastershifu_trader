function updateCooldownDisplay() { const cooldownInfo = document.getElementById('cooldownInfo'); const now = new Date(); const lastExitTime = tradingState.lastExitTime; const cooldownMs = 3000; if (lastExitTime && (now - lastExitTime) < cooldownMs) { cooldownInfo.style.display = 'block'; } else { cooldownInfo.style.display = 'none'; } }
