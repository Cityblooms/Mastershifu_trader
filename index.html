GitHub Actions Auto-Update Setup

Automatically update your application with CI/CD pipelines using GitHub Actions. This setup includes automated testing, building, and deployment.

1

Create Workflow File

Create a .github/workflows/auto-update.yml file in your repository.

auto-update.yml 
# GitHub Actions workflow for auto-updating application
name: Auto Update & Deploy
on:
  push:
    branches: [ main, master ]
  schedule:
    # Run every day at 5 AM UTC to check for updates
    - cron: '0 5 * * *'

jobs:
  update-and-deploy:
    runs-on: ubuntu-latest
    
    steps:
    - name: Checkout repository
      uses: actions/checkout@v3
      
    - name: Set up Node.js
      uses: actions/setup-node@v3
      with:
        node-version: '18'
        
    - name: Install dependencies
      run: npm ci
      
    - name: Run tests
      run: npm test
      
    - name: Build application
      run: npm run build
      
    - name: Deploy to production
      uses: JamesIves/github-pages-deploy-action@v4
      with:
        branch: gh-pages
        folder: dist

Auto-Update Client Script

Client-side JavaScript to check for updates and notify users.

update-checker.js 
// Client-side update checker
class UpdateChecker {
  constructor() {
    this.currentVersion = '1.2.0';
    this.updateEndpoint = 'https://api.your-app.com/version';
    this.checkInterval = 6 * 60 * 60 * 1000; // 6 hours
  }

  async checkForUpdates() {
    try {
      const response = await fetch(this.updateEndpoint);
      const data = await response.json();
      
      if (data.version !== this.currentVersion) {
        this.showUpdateNotification(data);
        return true;
      }
      return false;
    } catch (error) {
      console.error('Update check failed:', error);
      return false;
    }
  }

  showUpdateNotification(updateData) {
    // Create a premium-looking update notification
    const notification = document.createElement('div');
    notification.className = 'update-notification';
    notification.innerHTML = `
      <div class="update-content">
        <h3>✨ New Update Available! <span class="version">v${updateData.version}</span></h3>
        <p>${updateData.changelog || 'Bug fixes and performance improvements'}</p>
        <div class="update-actions">
          <button class="update-now">Update Now</button>
          <button class="update-later">Later</button>
        </div>
      </div>
    `;
    
    document.body.appendChild(notification);
    
    // Add event listeners for buttons
    notification.querySelector('.update-now').addEventListener('click', () => {
      this.applyUpdate();
    });
    
    notification.querySelector('.update-later').addEventListener('click', () => {
      notification.remove();
    });
    
    // Auto-remove after 24 hours
    setTimeout(() => {
      if (document.body.contains(notification)) {
        notification.remove();
      }
    }, 24 * 60 * 60 * 1000);
  }

  applyUpdate() {
    // Clear cache and reload for update
    if ('caches' in window) {
      caches.keys().then(function(cacheNames) {
        cacheNames.forEach(function(cacheName) {
          caches.delete(cacheName);
        });
      });
    }
    
    // Force reload from server
    window.location.reload(true);
  }

  startPeriodicCheck() {
    // Check immediately on load
    this.checkForUpdates();
    
    // Then check periodically
    setInterval(() => {
      this.checkForUpdates();
    }, this.checkInterval);
  }
}

// Initialize update checker when page loads
document.addEventListener('DOMContentLoaded', () => {
  const updateChecker = new UpdateChecker();
  updateChecker.startPeriodicCheck();
});

payment-service.js 
// Payment processing service for subscription management
class PaymentService {
  constructor() {
    // Initialize with your payment processor
    this.stripe = Stripe('pk_live_your_publishable_key');
    this.apiUrl = 'https://api.your-app.com/payments';
  }

  // Create a subscription for a user
  async createSubscription(userId, planId, paymentMethodId) {
    try {
      const response = await fetch(`${this.apiUrl}/subscribe`, {
        method: 'POST',
        headers: {
          'Content-Type': 'application/json',
          'Authorization': `Bearer ${getAuthToken()}`
        },
        body: JSON.stringify({
          userId,
          planId,
          paymentMethodId
        })
      });

      const data = await response.json();
      
      if (data.success) {
        // Update user's subscription status in your app
        this.updateUserSubscription(userId, {
          status: 'active',
          planId: planId,
          currentPeriodEnd: data.currentPeriodEnd
        });
        
        return { success: true, data };
      } else {
        return { success: false, error: data.error };
      }
    } catch (error) {
      console.error('Subscription creation failed:', error);
      return { success: false, error: error.message };
    }
  }

  // Check subscription status
  async checkSubscriptionStatus(userId) {
    try {
      const response = await fetch(`${this.apiUrl}/status/${userId}`, {
        headers: {
          'Authorization': `Bearer ${getAuthToken()}`
        }
      });

      return await response.json();
    } catch (error) {
      console.error('Status check failed:', error);
      return { active: false, error: error.message };
    }
  }

  updateUserSubscription(userId, subscriptionData) {
    // Update local user state or send to backend
    localStorage.setItem(`user_${userId}_subscription`, 
      JSON.stringify(subscriptionData));
    
    // Dispatch event for UI updates
    window.dispatchEvent(new CustomEvent('subscription-updated', {
      detail: { userId, subscription: subscriptionData }
    }));
  }
}

// Helper function to get auth token
function getAuthToken() {
  return localStorage.getItem('auth_token') || '';
}
